91 full_stops = train_service.get_full_stop_list() 

92 start_index = next((order for order, st in full_stops if st == from_station), None)

96 return Response({"error": "Invalid station sequence"}, status=400)

100 segment_index__lt=end_index 

101 ) 

102

104 class_prices, availability = {}, {} 

105 for class_type, field_name in self.CLASS_MAP.items():

108 class_prices[class_type] = str(price or Decimal("0.00")) 

109 except Exception: 

110 class_prices[class_type] = "0.00" 

111 

112 try: 

113 min_val = segments.aggregate(min_val=Min(field_name))["min_val"] 

114 availability[class_type] = int(min_val or 0) 

115 except Exception: 

116 availability[class_type] = 0
