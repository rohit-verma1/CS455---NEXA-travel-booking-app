15def send_otp_email(email, otp, otp_type="registration"): 

16 """ 

17 Validate email and send OTP. Returns status message string. 

18 """ 

19 

20 # 1) Validate format 

21 try: 

22 validate_email(email) 

23 except ValidationError: 

24 print(f"❌ Invalid email format: {email}") 

25 return f"Invalid email format: {email}" 

26 

27 # 2) Subject selection 

28 subject_map = { 

29 "registration": "Verify your email - OTP Code", 

30 "password_reset": "Password Reset OTP", 

31 "email_change": "Verify your new email address", 

32 } 

33 subject = subject_map.get(otp_type, "Your OTP Code") 

34 

35 # 3) Body 

36 text_content = f"Your OTP code is {otp}. It will expire in 10 minutes." 

37 html_content = f""" 

38 <div style="font-family:Arial,sans-serif;"> 

39 <h2>Your OTP Code</h2> 

40 <p style="font-size:18px;">Use the following OTP to complete your process:</p> 

41 <p style="font-size:24px;font-weight:bold;">{otp}</p> 

42 <p>This OTP will expire in 10 minutes.</p> 

43 <br> 

44 <p>Thanks,<br>The Booking Platform Team</p> 

45 </div> 

46 """ 

47 

48 # 4) Prepare message and send 

49 msg = EmailMultiAlternatives( 

50 subject=subject, 

51 body=text_content, 

52 from_email=getattr(settings, "DEFAULT_FROM_EMAIL", settings.EMAIL_HOST_USER), 

53 to=[email], 

54 ) 

55 msg.attach_alternative(html_content, "text/html") 

56 

57 # 5) Send email with error handling 

58 try: 

59 msg.send() 

60 print(f"✅ OTP email successfully sent to {email}") 

61 return "OTP successfully sent" 

62 except SMTPRecipientsRefused as e: 

63 print(f"❌ Invalid recipient: {email} - {e}") 

64 return f"Invalid recipient: {email}" 

65 except SMTPAuthenticationError as e: 

66 print(f"❌ SMTP authentication failed: {e}") 

67 return "SMTP authentication failed" 

68 except Exception as e: 

69 print(f"❌ Unexpected error while sending OTP to {email}: {e}") 

70 return "Unexpected error while sending OTP"
